macro(add_process_plugin_test test_name plugin_name pcap_file)
    add_test(
        NAME ${test_name}
        COMMAND bash ${CMAKE_CURRENT_SOURCE_DIR}/scripts/run_test.sh ${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_BINARY_DIR} ${plugin_name} ${pcap_file}
    )
endmacro()
